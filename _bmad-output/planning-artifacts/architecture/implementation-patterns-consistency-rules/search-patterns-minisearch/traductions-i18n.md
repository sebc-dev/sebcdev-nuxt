// i18n/locales/fr.json
{
  "search": {
    "label": "Rechercher",
    "placeholder": "Rechercher un article...",
    "loading": "Chargement...",
    "noResults": "Aucun résultat",
    "noResultsFor": "Aucun résultat pour \"{query}\"",
    "resultsCount": "{count} résultat(s) trouvé(s)",
    "articles": "Articles"
  }
}
